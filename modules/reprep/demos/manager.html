[docs]def all_demos(argv):
    r = Report('reprep_demos')
    for demo in DemoStorage.demo_list:
        ri = r.section(nid='%s' % demo.__name__,
                       caption=demo.__doc__
                       )
        demo(ri)

    r.to_html('reprep_demos_out/index.html')
