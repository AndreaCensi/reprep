[docs]def demo_columns(r, nplots, ncols):
    f = r.figure(cols=ncols)

    for k in range(nplots):
        with f.plot() as pylab:
            x = np.linspace(0, 2 * np.pi, 100)
            y = np.sin(x * (k + 1))
            pylab.plot(x, y)
